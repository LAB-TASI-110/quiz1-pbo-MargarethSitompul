    public static void main(String[] args) {
         Scanner input = new Scanner(System.in);

        System.out.print("Masukkan jumlah siswa: ");
        int N = input.nextInt();

        char[] jk = new char[N];
        int[] nilai = new int[N];

        for (int i = 0; i < N; i++) {
            System.out.println("\nSiswa ke-" + (i + 1));
            System.out.print("Jenis Kelamin (L/P): ");
            jk[i] = input.next().charAt(0);

            System.out.print("Nilai: ");
            nilai[i] = input.nextInt();
        }

        System.out.print("\nKode kelompok (1 = Laki-laki, 2 = Perempuan): ");
        int kode = input.nextInt();

        int total = 0;

        for (int i = 0; i < N; i++) {
            if (kode == 1 && (jk[i] == 'L' || jk[i] == 'l')) {
                total += nilai[i];
            } 
            else if (kode == 2 && (jk[i] == 'P' || jk[i] == 'p')) {
                total += nilai[i];
            }
        }

        System.out.println("Total nilai kelompok " + kode + " = " + total);

        input.close();
    }
    }
